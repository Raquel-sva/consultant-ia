Ta mission dans cette conversation :

- Lui présenter naturellement tes offres quand c'est pertinent
- L'amener à vouloir réserver un appel découverte de 30 min avec toi
- Si le prospect semble qualifié et intéressé, propose explicitement le RDV avec la phrase : "JE_PROPOSE_RDV"


Règles importantes :
- Tu parles à la 1ère personne (c'est toi Sophie qui réponds)
- Ton ton est professionnel, humain, direct — jamais de jargon creux
- Maximum 3-4 phrases par réponse

- Si le prospect te parle d'un problème concret, montre que tu as déjà aidé des cas similaires
- Quand tu détectes que le prospect est prêt pour un RDV, inclus exactement "JE_PROPOSE_RDV" dans ta réponse`;


let history = [];
let rdvShown = false;


const msgsEl = document.getElementById('msgs');


function addMsg(html, role, isTyping) {
  const d = document.createElement('div');
  d.className = 'msg ' + role + (isTyping ? ' typing' : '');
  d.innerHTML = html.replace(/\n/g, '<br>');
  msgsEl.appendChild(d);
  msgsEl.scrollTop = msgsEl.scrollHeight;
  return d;
}


function addQR(options) {
  const wrap = document.createElement('div');
  wrap.className = 'qr-wrap';
  options.forEach(opt => {
    const btn = document.createElement('button');
    btn.className = 'qr';
    btn.textContent = opt;
    btn.onclick = () => { wrap.remove(); sendText(opt); };
    wrap.appendChild(btn);
  });
  msgsEl.appendChild(wrap);
  msgsEl.scrollTop = msgsEl.scrollHeight;
}


function showRDV() {
  if (rdvShown) return;
  rdvShown = true;
  const card = document.createElement('div');
  card.className = 'rdv-card';
  card.innerHTML = `
    <div>
      <strong>Parlons de votre projet</strong>
      <p>Un appel de 30 min pour voir concrètement comment je peux vous aider. Sans engagement.</p>
    </div>
    <button class="rdv-btn" onclick="openRDV()">Réserver →</button>
  `;
  msgsEl.appendChild(card);
  msgsEl.scrollTop = msgsEl.scrollHeight;
}


function openRDV() {
  // Remplacez cette URL par votre lien Calendly
  window.open('https://calendly.com', '_blank');
}


async function sendText(txt) {
  addMsg(txt, 'u');
  history.push({ role: 'user', content: txt });
  await callAPI();
}


async function send() {
  const inp = document.getElementById('inp');
  const txt = inp.value.trim();
  if (!txt) return;
  inp.value = '';
  await sendText(txt);
}


async function callAPI() {
  const typing = addMsg('…', 'a', true);
  try {
    const res = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1000,
        system: SYSTEM,
        messages: history
      })
    });
    const data = await res.json();
    let reply = data.content.map(b => b.text || '').join('');
    const proposeRDV = reply.includes('JE_PROPOSE_RDV');
    reply = reply.replace(/JE_PROPOSE_RDV/g, '').trim();
    typing.remove();
    addMsg(reply, 'a');
    history.push({ role: 'assistant', content: reply });
    if (proposeRDV) showRDV();
  } catch(e) {
    typing.remove();
    addMsg("Une erreur s'est produite. Réessayez dans un instant.", 'a');
  }
}
